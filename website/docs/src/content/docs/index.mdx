import { Card, CardGrid, LinkCard } from '@astrojs/starlight/components';

## Why Puebla?

<CardGrid stagger>
  <Card title="Zero-JS by default" icon="rocket">
    Pages render to static HTML. React only hydrates the islands that need it.
  </Card>
  <Card title="File-based routing" icon="seti:html">
    `src/pages/*.astro` becomes `dist/<route>.html`. No `react-router`.
  </Card>
  <Card title="Salesforce-aware" icon="cloud-download">
    `@salesforce/vite-plugin-ui-bundle` rewrites asset URLs at runtime.
  </Card>
  <Card title="Tailwind v4" icon="puzzle">
    Modern styling with the latest Tailwind, plus design tokens.
  </Card>
</CardGrid>
